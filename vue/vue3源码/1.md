## Vue3新增了什么
• 性能 （⽐vue2快了2倍）
• tree shaking (按需编译代码)
• composition api （类似hooks未来）
• ts support （typescript ⾯向未来）
• custom renderer api （⾃定义渲染）

## Vue 2 缺点

Vue 2 中的option API 难以做tree shaking

`const { createApp, reactive, watchEffect, computed} = Vue`, Vue 3中如果没有用到computed，则打包的时候不会加载进来

## Proxy 响应式原理实现

```
<!DOCTYPE html>
<html>
<head>
    <title>vue3 reactive</title>
</head>
<body>
    <div id="app">
    </div>
    <button id="btn">add</button>
    <script type="text/javascript">
        // 缓存对象：原始-> 响应式
        let toProxy = new WeakMap()

        // 缓存对象：响应式 -> 原始
        let toRaw = new WeakMap()

        // 存储依赖关系
        let effectStack = [] // 存储effect
        let targetMap = new WeakMap()

        // 收集依赖
        function track (target,key) {
          // 获取最新的effect
          const effect = effectStack[effectStack.length -1]
          if (effect) {
            let depMap = targetMap.get(target)
            if (depMap === undefined) {
              depMap = new Map()
              targetMap.set(target, depMap)
            }
            let dep = depMap.get(key)
            if (dep == undefined) {
              dep = new Set()
              depMap.set(key, dep)
            }

            // 双向存储
            if (!dep.has(effect)) {
              dep.add(effect)
              effect.deps.push(dep)
            }
          }
        }

        // 触发更新
        function trigger (target, key, info) {
          // 寻找依赖effect
          const depMap = targetMap.get(target)
          if (depMap === undefined) {
            // 没有依赖
            return
          }
          const effects = new Set()
          const computedRunners = new Set() // computed是特殊的effect

          if (key) {
            let deps = depMap.get(key)
            // deps里全部是effect
            deps.forEach(effect => {
              if (effect.computed) {
                computedRunners.add(effect)
              } else {
                effects.add(effect)
              }
            })
          }
          effects.forEach(effect => effect())
          // computed依赖effect,所以后执行
          computedRunners.forEach(computed => computed())
        }

        function effect (fn, options= {}) {
          // 往effectStack里添加一个effect函数，并执行fn
          let e = createReactiveEffect(fn, options)
          if (!options.lazy) { // 不是lazy,首次先执行一次
            e()
          }
          return e
        }

        function createReactiveEffect (fn, options) {
          // 构造一个新的effect，真正执行的是传入的fn
          const effect = function effect(...args) {
            return run(effect, fn, args)
          }
          effect.deps = []
          effect.computed = options.computed
          effect.lazy = options.lazy
          return effect
        }

        function run (effect, fn, args) {
          if (effectStack.indexOf(effect) === -1) {
            try {
              effectStack.push(effect)
              return fn(...args) // 执行
            } finally{
              effectStack.pop()
            }
          }
        }

        function computed (fn) {
          const runner = effect(fn, {computed: true, lazy: true})
          return {
            effect: runner,
            get value () {
              return runner()
            }
          }
        }

        // 响应式代理对象
        const baseHandler = {
          get(target, key) {
            // target 为 obj, key 为 name
            // 搜集依赖 track
            const res = Reflect.get(target, key)
            track(target, key)
            return typeof res === 'object' ? reactive(res): res
          },
          set(target, key, val) {
            const info = {oldVal:target[key], newVal:val}
            const res = Reflect.set(target, key, val)
            // 触发更新 trigger
            trigger(target, key, info)
            return res
          }
        }
        
        // 响应式
        function reactive(target) {
          // 查询缓存,如果是已代理的对象，直接返回
          let observed = toProxy.get(target)
          if (observed) {
            return observed
          }

          if (toRaw.get(target)) {
            return target
          }

          observed = new Proxy(target, baseHandler)

          // 设置缓存
          toProxy.set(target, observed)
          toRaw.set(observed, target)
          return observed
        }
    </script>
    <script>
      const root = document.getElementById('app')
      const btn = document.getElementById('btn')

      const obj = reactive({
        name: 'money',
        age: 29
      })

      let double = computed(() => obj.age * 2)

      effect(() => {
        // 数据变化就执行
        console.log('change', obj.age)
        root.innerHTML = `<div>${obj.name} -- ${obj.age}</div>`
      })
      btn.addEventListener('click', () => {
        // 触发set 执行effect
        obj.age++
      })
    </script>
</body>
</html>
```
